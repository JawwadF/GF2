//Dtype device

// Devices

DEVICE = SWITCH VALUE 0 NAME set;
DEVICE = SWITCH VALUE 0 NAME clear;
DEVICE = SWITCH VALUE 0 NAME data;
DEVICE = SWITCH VALUE 0 NAME S4;

DEVICE = DTYPE NAME U1;
DEVICE = CLOCK CYCLES 2 NAME clock1;

// Connections
CONNECT = set > U1.SET;
CONNECT = clear > U1.CLEAR;
CONNECT = clock1 > U1.CLK;
CONNECT = data > U1.DATA;
//CONNECT = U1.QBAR > U3.I1;

// Monitors
MONITOR = RECORDS U1.Q NAME M1;
MONITOR = RECORDS U1.QBAR NAME M2;
