// Devices

DEVICE = AND INPUTS 2 NAME U2;
DEVICE = NAND INPUTS 3 NAME U3;

DEVICE = SWITCH VALUE 1 NAME S1;
DEVICE = SWITCH VALUE 1 NAME S2;
DEVICE = SWITCH VALUE 1 NAME S3;
DEVICE = SWITCH VALUE 0 NAME S4;

DEVICE = SIGGEN # NAME sgen;
DEVICE = SIGGEN #01010011 NAME sgen2;
DEVICE = DTYPE NAME U1;
DEVICE = CLOCK CYCLES 5 NAME clock1;

// Connections
CONNECT = S1 > U3.I2;
CONNECT = sgen > U2.I2;
CONNECT = S2 > U3.I3;
CONNECT = sgen2 > U2.I1;
CONNECT = S3 > U1.SET;
CONNECT = S4 > U1.CLEAR;
CONNECT = clock1 > U1.CLK;
CONNECT = U1.QBAR > U1.DATA;
CONNECT = U1.QBAR > U3.I1;

// Monitors
MONITOR = RECORDS sgen NAME monitor1;
MONITOR = RECORDS U1.Q NAME monitor2;
MONITOR = RECORDS sgen2 NAME monitor3;
MONITOR = RECORDS U2 NAME mon4;
